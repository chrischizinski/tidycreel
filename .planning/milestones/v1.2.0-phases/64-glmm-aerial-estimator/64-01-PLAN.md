Function signature:
```r
estimate_effort_aerial_glmm <- function(
  design,
  time_col,
  formula   = NULL,
  family    = NULL,
  boot      = FALSE,
  nboot     = 500L,
  conf_level = 0.95
)
```

Implementation order (follow exactly):

1. rlang::check_installed("lme4", reason = "to fit the GLMM aerial effort estimator")

2. design_type guard: if (!identical(design$design_type, "aerial")) cli_abort with found type and pointer to estimate_effort().

3. Resolve time_col: time_col_quo <- rlang::enquo(time_col); time_col_name <- rlang::as_name(time_col_quo). Validate column exists in design$counts.

4. Identify count variable: same logic as estimate_effort_aerial() — exclude date_col, strata_cols, psu_col, time_col_name from numeric columns. Take count_vars[1L]. Abort if none found.

5. Build glmm_formula: if (is.null(formula)) construct paste0(count_var, " ~ poly(", time_col_name, ", 2) + (1|", design$date_col, ")") as stats::as.formula(). Else use formula directly.

6. Fit model: if (is.null(family) || identical(family, "negbin")) use lme4::glmer.nb(glmm_formula, data = design$counts). Else use lme4::glmer(glmm_formula, data = design$counts, family = family).

7. Build prediction grid for integration: h_over_v <- design$aerial$h_open / (design$aerial$visibility_correction %||% 1.0). Integration window: open_start = min(design$counts[[time_col_name]]) - 0.5, open_end = max(design$counts[[time_col_name]]) + 0.5. hour_grid <- seq(open_start, open_end, length.out = 100). new_data with time_col_name column = hour_grid and date_col = NA_character_. Use stats::delete.response(stats::terms(model)) to get terms object. X <- stats::model.matrix(terms_obj, data = new_data). beta <- lme4::fixef(model). mu <- as.numeric(exp(X %*% beta)). scale_factor <- (open_end - open_start) / 100. total_effort <- sum(mu) * scale_factor * h_over_v.

8. Delta method SE (default, boot = FALSE): V <- as.matrix(stats::vcov(model)). grad <- scale_factor * h_over_v * colSums(mu * X). var_total <- as.numeric(t(grad) %*% V %*% grad). se <- sqrt(var_total). se_between <- se. alpha <- 1 - conf_level. z_crit <- stats::qnorm(1 - alpha / 2). ci_lower <- total_effort - z_crit * se. ci_upper <- total_effort + z_crit * se.

9. Bootstrap path (boot = TRUE): cli::cli_inform("Running {nboot} bootstrap replicates via lme4::bootMer..."). boot_fn <- function(m) { mu_b <- as.numeric(exp(X %*% lme4::fixef(m))); sum(mu_b) * scale_factor * h_over_v }. b <- lme4::bootMer(model, FUN = boot_fn, nsim = nboot, type = "parametric", use.u = FALSE). se <- stats::sd(b$t). ci_probs <- c((1 - conf_level) / 2, 1 - (1 - conf_level) / 2). ci_vec <- stats::quantile(b$t, ci_probs, names = FALSE). ci_lower <- ci_vec[1]; ci_upper <- ci_vec[2]. se_between <- se.

10. Assemble output: estimates_df <- tibble::tibble(estimate = total_effort, se = se, se_between = se_between, se_within = NA_real_, ci_lower = ci_lower, ci_upper = ci_upper, n = nrow(design$counts)). variance_method_str <- if (boot) "bootstrap" else "delta". new_creel_estimates(estimates = estimates_df, method = "aerial_glmm_total", variance_method = variance_method_str, design = design, conf_level = conf_level, by_vars = NULL).
