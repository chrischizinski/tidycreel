**8. Build design and run pipeline**:
```r
message("Building bus-route creel design...")
design <- creel_design(
  calendar       = cal,
  date           = date,
  strata         = day_type,
  survey_type    = "bus_route",
  sampling_frame = sf,
  site           = site,
  circuit        = circuit,
  p_site         = p_site,
  p_period       = p_period
)

message("Adding counts...")
design <- add_counts(design, cnt)

message("Adding interviews (duplicated UIDs preserved — bus-route design)...")
design <- add_interviews(
  design, ivw,
  catch         = catch_count,
  effort        = effort_hours,
  trip_status   = trip_status,
  n_counted     = n_counted,
  n_interviewed = n_interviewed
)

message("Running estimators...")
eff  <- suppressWarnings(estimate_effort(design))
cat_est <- suppressWarnings(estimate_total_catch(design))
harv <- suppressWarnings(estimate_total_harvest(design))
```

**9. Compare against reference outputs**:
```r
computed <- list(
  effort_total  = eff$estimates$estimate,
  catch_total   = cat_est$estimates$estimate,
  harvest_total = harv$estimates$estimate
)

REL_TOL <- 0.001  # 0.1% relative tolerance

results <- data.frame(
  estimand   = ref$estimand,
  reference  = ref$estimate,
  computed   = vapply(ref$estimand, function(nm) computed[[nm]], numeric(1L)),
  stringsAsFactors = FALSE
)
results$rel_error <- abs(results$computed - results$reference) / abs(results$reference)
results$pass      <- results$rel_error <= REL_TOL
```

**10. Print per-estimand results**:
```r
message("\n--- Calamus 2016 Validation Results ---")
for (i in seq_len(nrow(results))) {
  status <- if (results$pass[i]) "PASS" else "FAIL"
  message(sprintf("  [%s] %s: reference=%.4f  computed=%.4f  rel_error=%.6f",
                  status, results$estimand[i],
                  results$reference[i], results$computed[i], results$rel_error[i]))
}
```

**11. Overall result and exit code**:
```r
all_pass <- all(results$pass)
overall  <- if (all_pass) "PASS" else "FAIL"
message(sprintf("\n=== Overall: %s (%d/%d estimands within %.1f%% tolerance) ===",
                overall, sum(results$pass), nrow(results), REL_TOL * 100))

if (!all_pass) {
  failed <- results$estimand[!results$pass]
  message(sprintf("Failed estimands: %s", paste(failed, collapse = ", ")))
  quit(status = 1L)
}
invisible(results)
```

The script must complete without error. Use `suppressWarnings()` around estimate_* calls to suppress FPC-related survey package warnings (which are expected for small sample strata).

Do NOT call `add_catch()` anywhere in this script. The bus-route total catch estimator reads `catch_count` from the interviews data frame (which is the total catch per interview). The catch.csv table is referenced only in the data-logging section (step 5) to verify edge cases but is not fed into the estimation pipeline.
