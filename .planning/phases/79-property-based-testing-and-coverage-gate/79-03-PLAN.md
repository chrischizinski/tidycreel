test_that("INV-03 smoke: ice equals degenerate bus-route for fixed params", {
  set.seed(42)

# Property test for INV-03
test_that("INV-03: ice == degenerate bus-route for all random param sets", {
  for_all(
    params = gen.list(
      n_sites = gen.int(n = 9, min = 2),
      n_days  = gen.int(n = 15, min = 5),
      n_int   = gen.int(n = 9, min = 1)

      ice_result <- estimate_harvest_rate(
        build_ice_design(params$n_sites, params$n_days, params$n_int)

        build_br_degenerate_design(params$n_sites, params$n_days, params$n_int)
      )
      isTRUE(all.equal(ice_result$estimates, br_result$estimates, tolerance = 1e-6))

  )
