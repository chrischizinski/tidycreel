# Counter mode: 10-12 rows (8-12 sampling days)

# Valid camera_status values: "operational", "battery_failure", "memory_full", "occlusion"
example_camera_counts <- data.frame(
  date          = as.Date(c("2024-06-03", "2024-06-04", ...)),
  day_type      = c("weekday", "weekday", "weekend", ...),
  ingress_count = c(45L, 62L, NA_integer_, ...),  # NA on gap row
