```r
#' @export
fetch_counts.creel_connection_api <- function(conn, ...) {
  raw_df <- .api_fetch(conn$con, "counts")

  # Early return for empty API response — include n_counted/n_interviewed for schema consistency (D-10)
  if (nrow(raw_df) == 0L) {
    return(data.frame(
      date             = as.Date(character(0)),
      angler_count     = numeric(0),
      n_counted        = integer(0),
      n_interviewed    = integer(0),
      stringsAsFactors = FALSE
    ))
  }

  # Hardcoded NGPC field names — do NOT route through creel_schema (CONTEXT.md D-15)
  # Field names confirmed via ngpc-field-discovery.R probe (API-09, API-11)
  api_rename_map <- c(
    date          = "cd_Date",
    angler_count  = "ii_NumberAnglers",  # confirmed NGPC field name
    n_counted     = "[N_COUNTED_FIELD]", # confirmed via ngpc-field-discovery.R
    n_interviewed = "[N_INTERVIEWED_FIELD]" # confirmed via ngpc-field-discovery.R
  )
  df <- .rename_api_to_canonical(raw_df, api_rename_map)

  if ("date" %in% names(df))         df$date         <- .parse_api_date(df$date)
  if ("angler_count" %in% names(df)) df$angler_count  <- as.numeric(df$angler_count)

  # NA backfill: non-bus-route API responses lack n_counted/n_interviewed (CONTEXT.md D-09)
  # .rename_api_to_canonical() silently drops absent columns; backfill ensures consistent structure
  if (!"n_counted" %in% names(df)) {
    df$n_counted <- NA_integer_
  }
  if (!"n_interviewed" %in% names(df)) {
    df$n_interviewed <- NA_integer_
  }

  validate_fetch_counts(df) # nolint: object_usage_linter
  df
}
```
