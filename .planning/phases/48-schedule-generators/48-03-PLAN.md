write_schedule():
```r
#' Write a schedule to CSV or xlsx
#' @param schedule A creel_schedule tibble.
#' @param path File path (extension determines format if format not supplied).
#' @param format One of "csv" (default) or "xlsx".
#' @return Invisible path.
#' @export
write_schedule <- function(schedule, path, format = c("csv", "xlsx")) {
  format <- match.arg(format)
  if (format == "xlsx") {
    rlang::check_installed("writexl", reason = "to write xlsx schedule files")
    writexl::write_xlsx(schedule, path)
  } else {
    utils::write.csv(schedule, path, row.names = FALSE)
  }
  invisible(path)
}
```

read_schedule():
```r
#' Read a schedule file into a validated creel_schedule object
#' @param path Path to a CSV or xlsx schedule file.
#' @return A validated `creel_schedule` tibble.
#' @export
read_schedule <- function(path) {
  ext <- tolower(tools::file_ext(path))
  if (ext %in% c("xlsx", "xls")) {
    rlang::check_installed("readxl", reason = "to read xlsx schedule files")
    raw <- readxl::read_excel(path, col_types = "text")
  } else {
    raw <- utils::read.csv(path, stringsAsFactors = FALSE, colClasses = "character")
  }
  raw <- as.data.frame(raw)  # ensure plain data.frame for coercion
  coerced <- coerce_schedule_columns(raw)
  validate_creel_schedule(coerced)
  new_creel_schedule(coerced)
}
```
