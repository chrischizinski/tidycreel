# GLM method ----

test_that("CAMP-01a: returns complete frame with no NA in count_col", {
  counts <- make_camera_counts_with_outage()

  expect_true(all(!is.na(result$ingress_count)))

test_that("CAMP-01b: .imputed is TRUE for outage rows", {
  counts <- make_camera_counts_with_outage()
  result <- impute_camera_counts(counts,
                                  count_col = "ingress_count",
                                  strata_col = "day_type")
  expect_true(".imputed" %in% names(result))
  outage_row <- counts$camera_status != "operational" & is.na(counts$ingress_count)
  expect_true(all(result$.imputed[outage_row]))

test_that("CAMP-01c: .imputed is FALSE for operational rows", {
  counts <- make_camera_counts_with_outage()
  result <- impute_camera_counts(counts,
                                  count_col = "ingress_count",

  operational_row <- counts$camera_status == "operational"
  expect_true(all(!result$.imputed[operational_row]))

test_that("CAMP-02: method = 'glmm' proceeds or errors on missing glmmTMB", {
  counts <- make_camera_counts_with_outage()
  # This test checks the guard fires; actual GLMM fit depends on glmmTMB availability.
  # If glmmTMB is installed (CI env may have it), result should be a data frame.
  # If not installed, expect an error from rlang::check_installed().
  if (requireNamespace("glmmTMB", quietly = TRUE)) {
    result <- impute_camera_counts(counts,
                                    count_col = "ingress_count",
                                    strata_col = "day_type",
                                    method = "glmm")
    expect_s3_class(result, "data.frame")
    expect_true(".imputed" %in% names(result))
  } else {
    expect_error(
      impute_camera_counts(counts,
                            count_col = "ingress_count",
                            strata_col = "day_type",
                            method = "glmm"),
      regexp = "glmmTMB"
    )
  }
})
